    predict = np.random.randint(1,101)

def score_game(game_core):

    count_ls = []
    np.random.seed(1)  # фиксируем RANDOM SEED, чтобы ваш эксперимент был воспроизводим!

    for number in random_array:

    score = int(np.mean(count_ls))
